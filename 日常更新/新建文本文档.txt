#include <stdio.h>

int main() {
    int h1, r1;
    scanf("%d,%d", &h1, &r1);
     double h = h1 / 100.0;
     double r = r1 / 100.0;//单位换算成m

    // 计算每桶的体积
    double v = 3.14159 * r * r * h *1000;//1m^3 = 1000L
    // 计算君君要提的桶数
    int b = (20.0 / v) + 1; // 向上取整
    printf("%d\n", b);

    return 0;
}
